/* CORE GOOGLE COMMUNICATIONS TELEMETRY */ async function fetchLogs() { const tbody = document.getElementById('liveFeedBody'); try { const response = await fetch(API_URL); const data = await response.json(); tbody.innerHTML = ''; if(data.length === 0) { tbody.innerHTML = 'No entries recorded today.'; return; } document.getElementById('metric-total').innerText = data.length; let inboundCount = data.filter(item => item.direction === 'Inbound').length; document.getElementById('metric-inbound').innerText = inboundCount; let uniqueDocks = [...new Set(data.map(item => item.dockNumber))].length; document.getElementById('metric-docks').innerText = uniqueDocks; data.reverse().forEach(row => { const tr = document.createElement('tr'); const badgeClass = row.direction.toLowerCase() === 'inbound' ? 'in' : 'out'; tr.innerHTML = '' + row.driverName + '' + row.plateNumber + '' + row.dockNumber + '' + row.direction + '' + row.securityName + ''; tbody.appendChild(tr); }); } catch (error) { console.error("Error fetching data:", error); // Displays the clear error message directly inside the table frame tbody.innerHTML = '⚠️ Connection Error: Check Google Web App Permissions (Set to "Anyone").'; } }
